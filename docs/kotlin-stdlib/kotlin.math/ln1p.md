

# ln1p

Computes ln(x + 1).

```kotlin
expect fun ln1p(x: Double): Double(source)
```

import kotlin.math.ln1p

fun main() {
    val x = 0.5
    val result = ln1p(x)
    println("ln($x + 1) = $result")   // prints: ln(0.5 + 1) = 0.4054651081081644
}
